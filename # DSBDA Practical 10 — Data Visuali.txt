# Complete Practical Code

```python
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Load dataset

df = sns.load_dataset('iris')

# Display dataset
print(df.head())

# Dataset info
print(df.info())

# Columns
print(df.columns)

# Data types
print(df.dtypes)

# Histograms

df.hist(figsize=(10,8))
plt.show()

# Single Histogram

df['sepal_length'].hist()
plt.title('Sepal Length Distribution')
plt.show()

# Boxplot for all features

plt.figure(figsize=(10,6))
sns.boxplot(data=df)
plt.title('Boxplot for Iris Features')
plt.show()

# Single Boxplot

sns.boxplot(y=df['sepal_width'])
plt.title('Sepal Width Boxplot')
plt.show()
```
