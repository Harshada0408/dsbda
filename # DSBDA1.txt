# Complete Practical Code (Single File)

```python
# Import Libraries
import pandas as pd
import numpy as np

# Load Dataset
# Make sure Titanic-Dataset.csv is in same folder

df = pd.read_csv('Titanic-Dataset.csv')

# Display first rows
print("
First 5 Rows:")
print(df.head())

# Dataset Shape
print("
Dataset Shape:")
print(df.shape)

# Column Names
print("
Column Names:")
print(df.columns)

# Data Types
print("
Data Types:")
print(df.dtypes)

# Dataset Information
print("
Dataset Info:")
print(df.info())

# Statistical Summary
print("
Statistical Summary:")
print(df.describe())

# Include Categorical Columns
print("
Full Summary:")
print(df.describe(include='all'))

# Missing Values
print("
Missing Values:")
print(df.isnull().sum())

# Fill Missing Age Values
# Replace null age values using mean

df['Age'] = df['Age'].fillna(df['Age'].mean())

# Check Data Types Again
print("
Updated Data Types:")
print(df.dtypes)

# Convert Survived to Category Type

df['Survived'] = df['Survived'].astype('category')

# Convert Gender to Numeric

df['Sex'] = df['Sex'].map({'male':0, 'female':1})

# Convert Embarked to Numeric

df['Embarked'] = df['Embarked'].map({'S':0, 'C':1})

# Display Updated Dataset
print("
Updated Dataset:")
print(df.head())
```
