# cell 2 – importing required libraries

# importing required python libraries

# cell 12 – selecting input and output variables

# selecting independent variables

x = df.drop('MEDV', axis=1)

# selecting dependent variable

y = df['MEDV']

# displaying input variables

print(x.head())

# displaying output variable

print(y.head())


---

# cell 13 – splitting dataset into training and testing data

# splitting dataset into training and testing sets

x_train, x_test, y_train, y_test = train_test_split(
    x,
    y,
    test_size=0.2,
    random_state=42
)

# displaying shapes

print(x_train.shape)
print(x_test.shape)
print(y_train.shape)
print(y_test.shape)


---

# cell 14 – creating linear regression model

# creating linear regression model

model = LinearRegression()


---

# cell 15 – training the model

# training linear regression model

model.fit(x_train, y_train)

# model learns relationship between features and house price


---

# cell 16 – predicting house prices

# predicting values using testing data

y_pred = model.predict(x_test)

# displaying predicted values

print(y_pred)


---

# cell 17 – comparing actual and predicted values

# creating dataframe for comparison

comparison = pd.DataFrame({
    'actual_price': y_test,
    'predicted_price': y_pred
})

# displaying comparison

print(comparison.head(10))


---

# cell 18 – calculating model accuracy metrics

# calculating mean absolute error

mae = mean_absolute_error(y_test, y_pred)

# calculating mean squared error

mse = mean_squared_error(y_test, y_pred)

# calculating root mean squared error

rmse = np.sqrt(mse)

# calculating r2 score

r2 = r2_score(y_test, y_pred)

# displaying results

print("mean absolute error:", mae)

print("mean squared error:", mse)

print("root mean squared error:", rmse)

print("r2 score:", r2)

# evaluation metrics

---

# mean absolute error

measures average absolute difference between actual and predicted values.

formula:

MAE = (1/n)*​(∑|yi​−yi^|)

smaller mae means better model.

---

# mean squared error

measures squared difference between actual and predicted values.

formula:

MSE= (1/n)*​(∑(yi​−yi^​)^2)

n: Total number of observations (data points).
∑: Summation symbol (summing from i=1 to n).
yi): The actual observed value.
yi^): The predicted value.
(yi - yi^)^2: The squared difference (residual) between actual and predicted values.
---

# root mean squared error

square root of mse.

formula:

RMSE = sqrt(MSE)

---

# r2 score

measures goodness of fit of model.

range:

* closer to 1 = better model
* closer to 0 = poor model
